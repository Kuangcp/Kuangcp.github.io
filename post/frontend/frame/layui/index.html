    function string(){
        var set_content = $("#set_key").html()
        layer.tab({
            area: ['500px', '520px'],
            tab: [{
                title: 'set', 
                content: set_content
            }]
        });        
        $("#set-form").submit(function(e){
            e.preventDefault();
            set();
        });
    }
    function set(){
        var key = $("#key").val()
        var value = $("#value").val()
        handlePost('/key',  {
            key: key, 
            value: value
        }, function(data){
            console.log(data)
        }, function(data){
            console.log(data)
        })
    }
}
